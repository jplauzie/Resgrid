int flipped_count = 0;
bool factor_is_stale = true;

for (int step = 0; step <= total_steps; step++) {
    // 1. Update Rgrid and count changes
    int new_flips = 0;
    for(int i=0; i<X*Y; i++) {
        if(temp >= Tgrid_up[i] && Rgrid[i] > 1.0) { Rgrid[i] = 1.0; new_flips++; }
    }
    flipped_count += new_flips;

    // 2. Decide: Factorize or Iterative Solve?
    if (flipped_count > 1000 || factor_is_stale) {
        updateAx((double*)A->x, ...);
        cholmod_l_factorize(A, L, &c); // The expensive part
        cholmod_l_solve2(CHOLMOD_A, L, b, ...); // Use result directly
        flipped_count = 0;
        factor_is_stale = false;
        printf(" [Factorized] ");
    } else {
        updateAx((double*)A->x, ...);
        int iters = solve_pcg(A, L, b, x, &c); // The lightning fast part
        printf(" [PCG: %d iters] ", iters);
    }
}
